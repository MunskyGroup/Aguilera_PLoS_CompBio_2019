1function X_output = SSA_longNames(k,t_array,timePerturbationApplication,evaluatingInhibitor,evaluatingFRAP)
 2%% Defining parameters
 3exclusion = 9;                      % exclusion volume
 4maximum_Number_Ribosomes = 50;       % maximum number of ribosomes
 5k_elongation = k(2:end-1);           % elongation constants
 6k_bind = k(1);                       % Initiation rate
 7k_termination = k(end);              % Termination rate.
 8gene_Length = length(k_elongation);  % number of codons.
 9X_State = zeros(maximum_Number_Ribosomes,1);                       % Initial state (length of the longest protein = 1)
10t = t_array(1);                      % time array
11%% Preallocating memory
12number_TimePoints = length(t_array);
13t_final = t_array(number_TimePoints);
14X_output = zeros(maximum_Number_Ribosomes,number_TimePoints);
15
16%% Run SSA
17iteration = 1;
18while t < t_final
19    
20    if t >= timePerturbationApplication && evaluatingInhibitor ==1
21        Inhibitor_Condition = 0; % Set to 0 during its application
22    else
23        Inhibitor_Condition = 1; % Set to 1 before its application
24    end
25    if evaluatingFRAP ==1 && t >= timePerturbationApplication && t <= timePerturbationApplication+10
26        X_State = zeros(maximum_Number_Ribosomes,1);
27    end
28    %% Compute propensity functions
29    Nribosomes = sum(X_State > 0);
30    X_State(Nribosomes + 1,1) = 0;
31    temp_Stoichiometry = eye(Nribosomes + 1);
32    temp_Propensities = zeros(Nribosomes + 1,1);
33    temp_Propensities(X_State > 0,1) = k_elongation(X_State(X_State > 0));
34    
35    % termination
36    if X_State(1) >= gene_Length
37        temp_Stoichiometry(:,1) = [X_State(2:Nribosomes+1);0] - X_State(1:Nribosomes+1);
38        temp_Propensities(1,1) = k_termination;
39    end
40    % initiation
41    if Nribosomes == 0 || X_State(Nribosomes) > exclusion
42        temp_Propensities(Nribosomes+1,1) = k_bind * Inhibitor_Condition;
43    end
44    % elongation
45    elongation_Condition = (X_State(2:Nribosomes) + exclusion) < X_State(1:Nribosomes-1);
46    temp_Propensities(2:Nribosomes,1) = temp_Propensities(2:Nribosomes,1) .* elongation_Condition;
47    
48    % deffining prop and stoich
49    propensities = temp_Propensities;
50    stoichiometry = temp_Stoichiometry;
51    %% Update time
52    sum_Propensities = sum(propensities);
53    tau =  -log(rand) / sum_Propensities;
54    
55    if (evaluatingInhibitor==1) && (t<timePerturbationApplication) && ((t+tau)>timePerturbationApplication)
56        t = timePerturbationApplication;
57        %% Generate output
58        while iteration <= number_TimePoints && t > t_array(iteration)
59            X_output(1:size(X_State,1),iteration) = X_State;
60            iteration = iteration + 1;
61        end
62    else
63        t = t + tau;
64        %% Generate output
65        while iteration <= number_TimePoints && t > t_array(iteration)
66            X_output(1:size(X_State,1),iteration) = X_State;
67            iteration = iteration + 1;
68        end
69        %% Update state
70        if t < t_final
71            %% Select reaction
72            r2 = sum_Propensities * rand;
73            i = 1;
74            tmp = propensities(i);
75            while tmp < r2
76                i = i + 1;
77                tmp = tmp + propensities(i);
78            end
79            %% update state
80            X_State(1:Nribosomes+1) = X_State(1:Nribosomes+1) + stoichiometry(:,i);
81        end
82        
83    end
84end
